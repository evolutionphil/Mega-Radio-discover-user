// api/GenreService.js
class GenreService {
  /**
   * Get all genres globally (no country filter)
   */
  async getGlobalGenres() {
    try {
      const response = await fetch(
        'https://themegaradio.com/api/genres?tv=1&limit=100&sortColumn=stationCount&sortOrder=desc',
        {
          method: 'GET',
          headers: {
            'User-Agent': 'Tizen/6.0 Samsung TV',
            'Accept': 'application/json'
          }
        }
      );
      const data = await response.json();
      
      console.log(`‚úÖ Loaded ${data.genres.length} global genres`);
      
      return data.genres;
    } catch (error) {
      console.error('‚ùå Failed to fetch genres:', error);
      throw error;
    }
  }
  /**
   * Get genres for specific country
   */
  async getCountryGenres(countryName) {
    const url = `https://themegaradio.com/api/genres?tv=1&countrycode=${encodeURIComponent(countryName)}&limit=100`;
    
    const response = await fetch(url, {
      method: 'GET',
      headers: {
        'User-Agent': 'Tizen/6.0 Samsung TV',
        'Accept': 'application/json'
      }
    });
    const data = await response.json();
    return data.genres;
  }
}
